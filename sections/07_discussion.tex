\begin{frame}{Масштаб исследования}
	\begin{itemize}
		\item Дата-центр Acme: 4\,704 GPU A100
		\item 1.09 млн LLM-задач за 6 месяцев
		\item Среднее число GPU на задачу - 6.3
		\item Модели от 7B до 123B параметров
		\item Анализ этапа обучения (serving вынесен в отдельный кластер)
	\end{itemize}
		
	\vspace{0.3cm}
	\textbf{Важно:} исследование основано на промышленной эксплуатации,
	а не на лабораторном эксперименте.
\end{frame}
	
\begin{frame}{Выявленные системные проблемы}
	\begin{itemize}
		\item Падение throughput в 2--3 раза из-за Garbage Collection
		\item До 30\% времени шага занимала функция \texttt{list\_traverse}
		\item Утечка памяти в PyTorch Dataloader -> сбой в среднем через 27 часов
		\item Низкая утилизация GPU в MoE на 1024 GPU
		\item Узкое место - all-to-all сетевые коммуникации
	\end{itemize}
	\vspace{0.3cm}
	\textbf{Вывод:} LLM-нагрузки создают новые инфраструктурные риски.
\end{frame}
