\begin{frame}{Сбои и их влияние}
	\textbf{Частые ранние ошибки:}
	\begin{itemize}
		\item Множество ошибок возникает в начале исполнения задач LLM;
		\item Приводят к быстрому завершению заданий;
		\item Требуют оперативной диагностики для повышения эффективности обучения.
	\end{itemize}
	
	\textbf{Инфраструктурные сбои при предобучении:}
	\begin{itemize}
		\item Сбои в долгих предобучающих задачах значительно снижали эффективность обучения;
		\item Автоматизация восстановления становится критически важной.
	\end{itemize}
\end{frame}
